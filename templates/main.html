// Java Question:
                // Create a recursive method in Java to find the sum of the first n natural numbers.
                // Then, use this method to calculate the sum of the first 5 natural numbers and print the result.
                public class Main {
                    public static int sumOfNaturalNumbers(int n) {
                        if (n == 0)
                            return 0;
                        else
                            return n + sumOfNaturalNumbers(n - 1);
                    }
                
                    public static void main(String[] args) {
                        int n = 5;
                        System.out.println("Sum of first " + n + " natural numbers is: " + sumOfNaturalNumbers(n));
                    }
                }
